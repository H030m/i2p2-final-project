        if (!enemy.alive) {

                enemy.alive = true;
                enemy.cooldownTimer = 0;
                enemy.x = enemy.spawnX * TILE_SIZE;
                enemy.y = enemy.spawnY * TILE_SIZE;

        // 找到最近的玩家位置

Engine::Point RenderSender::nextPosition(float ex, float ey, Engine::Point target) {
    int gx = int(ex / TILE_SIZE);
    int gy = int(ey / TILE_SIZE);
    int tx = int(target.x / TILE_SIZE);
    int ty = int(target.y / TILE_SIZE);

    const int H = storedMapState["MapHeight"];
    const int W = storedMapState["MapWidth"];
    std::queue<std::pair<int, int>> q;
    std::vector<std::vector<Engine::Point>> prev(H, std::vector<Engine::Point>(W, {-1, -1}));
    q.push({gx, gy});

    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};

    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        if (x == tx && y == ty) break;

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H &&
                storedMapState["MapState"][ny][nx]["Block"] != 1 &&
                prev[ny][nx].x == -1) {
                prev[ny][nx] = {x, y};
                q.push({nx, ny});
            }
        }
    }

    // 從目標回溯一格
    int cx = tx, cy = ty;
    if (prev[cy][cx].x == -1) return {ex, ey}; // 無法到達
    while (!(prev[cy][cx].x == gx && prev[cy][cx].y == gy)) {
        auto p = prev[cy][cx];
        cx = p.x; cy = p.y;
    }

    return Engine::Point(cx * TILE_SIZE, cy * TILE_SIZE);
}

RenderSender::Point RenderSender::nextPosition(float ex, float ey, Point target) {
    int gx = int(ex / TILE_SIZE);
    int gy = int(ey / TILE_SIZE);
    int tx = int(target.x / TILE_SIZE);
    int ty = int(target.y / TILE_SIZE);

    const int H = storedMapState["MapHeight"];
    const int W = storedMapState["MapWidth"];
    std::queue<std::pair<int, int>> q;
    std::vector<std::vector<Point>> prev(H, std::vector<Point>(W, Point(-1, -1)));
    q.push({gx, gy});

    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};

    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        if (x == tx && y == ty) break;

        for (int d = 0; d < 4; ++d) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H &&
                storedMapState["MapState"][ny][nx]["Block"] != 1 &&
                prev[ny][nx].x == -1) {
                prev[ny][nx] = Point(x, y);
                q.push({nx, ny});
            }
        }
    }

    // 回推一格作為 next position
    int cx = tx, cy = ty;
    if (prev[cy][cx].x == -1) return Point(ex, ey); // 無法到達
    while (!(prev[cy][cx].x == gx && prev[cy][cx].y == gy)) {
        auto p = prev[cy][cx];
        cx = p.x; cy = p.y;
    }

    return Point(cx * TILE_SIZE, cy * TILE_SIZE);
}

void RenderSender::initializeEnemiesFromMap() {
    enemies.clear();
    if (!storedMapState.has_value()) return;

    auto& map = storedMapState.value();
    int H = map["MapHeight"];
    int W = map["MapWidth"];
    const auto& grid = map["MapState"];

    for (int y = 0; y < H; ++y) {
        for (int x = 0; x < W; ++x) {
            if (grid[y][x].contains("SpawnPoint")) {
                int type = grid[y][x]["SpawnPoint"];
                EnemyInstance enemy;
                enemy.id = nextEnemyId++;
                enemy.type = type;
                enemy.spawnX = x;
                enemy.spawnY = y;
                enemy.x = x * TILE_SIZE;
                enemy.y = y * TILE_SIZE;
                enemies.push_back(enemy);
            }
        }
    }

    std::cout << "Initialized " << enemies.size() << " enemies from map.\n";
}
